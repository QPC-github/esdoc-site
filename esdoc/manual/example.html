    Integrate Test Code Into Documentation

    {
  "source": "./src"
  "destination": "./doc",
  "test": {
    "type": "mocha",
    "source": "./test"
  }
}

    And if use @test, more better integration.

    /** @test {MyClass} */
describe('MyClass has foo bar feature', ()=>{

  /** @test {MyClass#baz} */
  it('MyClass#baz returns magic value', ()=>{
    assert(true);
  });
});
